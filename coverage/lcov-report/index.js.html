import axios from "axios";
import "whatwg-fetch";
import Config from "Config";
 
export function searchCommand(query) {
  return (dispatch) => {
    let promise = fetch(`http://localhost:8081/search?q=${query}`).then((response) => {
      return response.json();
    });
 
    dispatch({
      type: "SEARCH_COMMAND",
      payload: promise
    }).catch(error => {
      console.log(error)
    });;
 
    return promise;
  };
}
 
export function getCommands() {
  return (dispatch) => {
 
    let promise = fetch(`${Config.apiUrl}list`).then((response) => {
      return response.json();
    });
 
    dispatch({
      type: "GET_COMMAND",
      payload: promise
    });
    return promise;
  };
}
 
export function selectCommand(id) {
  return (dispatch) => {
    dispatch({ type: "SELECT_COMMAND", payload: id });
  };
}
export function showAddForm() {
  return (dispatch) => {
    dispatch({ type: "SHOW_ADD_FORM" });
  };
}
export function hideAddForm() {
  return (dispatch) => {
    dispatch({ type: "HIDE_ADD_FORM" });
  };
}
 
export function addCommand(command) {
  return (dispatch) => {
    var promise = axios.post(`${Config.apiUrl}add`, command);
    dispatch({ type: "ADD_COMMAND", payload: promise })
    return promise
  }
}
 
export function editCommand(command) {
  return (dispatch) => {
    dispatch({ type: 'SHOW_EDIT_FORM', payload: command })
  }
}
export function updateCommand(command) {
  return (dispatch) => {
    dispatch({ type: 'UPDATE_COMMAND', payload: axios.post(`${Config.apiUrl}update`, command) })
  }
}
 
export function deleteCommand(id) {
  return (dispatch) => {
    dispatch({
      type: 'DELETE_COMMAND',
      payload: axios.post(`${Config.apiUrl}delete`, { _id: id })
    })
  }
}
